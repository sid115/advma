\begin{exercise}
	Determine the volume $V$ of the portion of a ball of radius 3 that is defined by polar angles $\theta \in \left[0, \frac{\pi}{6}\right]$ and arbitrary azimuth angles $\varphi$.

	\emph{Remark:} $V$ is called a spherical sector or a spherical cone.
\end{exercise}

\begin{solution}
	$$
		\begin{aligned}
			V & = \iiint_{[0, 3] \times [0, \tfrac{\pi}{6}] \times [-\pi, \pi]} \, dV                       \\
			  & = \int_{0}^{2\pi} \int_{0}^{\pi/6} \int_{0}^{3} r^2 \sin\theta \, dr \, d\theta \, d\varphi \\
			  & = 2\pi \int_{0}^{\pi/6} \int_{0}^{3} r^2 \sin\theta \, dr \, d\theta                        \\
			  & = 2\pi \left[ \tfrac{1}{3} r^3 \right]_{0}^{3} \int_{0}^{\pi/6} \sin\theta \, d\theta       \\
			  & = -18\pi \left[ \cos\theta \right]_{0}^{\pi/6}                                              \\
			  & = 9\pi ( 2 - \sqrt{3} )
		\end{aligned}
	$$
\end{solution}
