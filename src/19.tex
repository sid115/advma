\begin{exercise}
	Compute the volume between the plane defined by
	$$
		z = 2x + 3y
	$$
	and the triangle
	$$
		D = \{(x,y) \mid x \in [1,2] \wedge y \in [0,x]\}
	$$
	in the $xy$-plane.
\end{exercise}

\begin{solution}
	$$
		\begin{aligned}
			V & = \iint_D z \, dA                                                 \\
			  & = \int_1^2 \int_0^x (2x + 3y) \, dy \, dx                         \\
			  & = \int_1^2 \left[ 2xy + \frac{3}{2} y^2 \right]_{y=0}^{y=x} \, dx \\
			  & = \int_1^2 ( 2x^2 + \frac{3}{2} x^2 ) \, dx                       \\
			  & = \left[ \frac{2}{3} x^3 + \frac{1}{2} x^3 \right]_1^2            \\
			  & = \left[ \frac{7}{6} x^3 \right]_1^2                              \\
			  & = \frac{56}{6} - \frac{7}{6}                                      \\
			  & = \frac{49}{6}
		\end{aligned}
	$$
\end{solution}
